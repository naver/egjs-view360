PanoViewer

Youtube (EAC Format)

EAC Format(Youtube) is supported also.

projection: CUBESTRIP

cubemapConfig: {order: "BLFDRU",tileConfig: [...]}

source: 

var PanoViewer = eg.view360.PanoViewer;
var container = document.getElementById("myPanoViewer");
var panoViewer = new PanoViewer(container, {
    image: "./img/youtube.png",
    projectionType: cubestrip/* PanoViewer.PROJECTION_TYPE.CUBESTRIP */,
   cubemapConfig: {
        order: "BLFDRU",
        tileConfig: [{rotation: 0}, {rotation: 0}, {rotation: 0}, {rotation: 0}, {rotation: -90}, {rotation: 180}]
    }
});
