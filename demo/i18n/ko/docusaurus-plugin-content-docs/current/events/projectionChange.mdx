import View360, { EVENTS } from "@site/src/components/View360";

```ts
import { EVENTS, ProjectionChangeEvent } from "@egjs/view360";

// typescript 코드입니다.
const onProjectionChange = (evt: ProjectionChangeEvent) => {};

view360.on(EVENTS.PROJECTION_CHANGE, onProjectionChange);
```

프로젝션이 변경되었을 때 발생하는 이벤트.

## 프로퍼티
### projection: [Projection](/docs/api/Projection/Class/Projection)
새로 적용된 프로젝션의 인스턴스

:::info
아래 예제는 이벤트 순서 확인을 위해 의도적으로 모든 컨트롤을 비활성화했습니다.

예제에서 확인 가능한 이벤트 목록은 다음과 같습니다.
- "loadStart", "load", "ready", "projectionChange"
:::

<View360
  projectionOptions={{
    src: "/pano/equirect/veste.jpg"
  }}
  license="veste"
  clickToLoad={true}
  rotate={false}
  zoom={false}
  gyro={false}
  showControlBar={false}
  showEvents={[EVENTS.LOAD_START, EVENTS.LOAD, EVENTS.READY, EVENTS.PROJECTION_CHANGE]} />
