PanoViewer

Setting the initial view

projection: EQUIRECTANGULAR

image: equi.jpg 

// create PanoViewer with option
var PanoViewer = eg.view360.PanoViewer;
var container = document.getElementById("myPanoViewer");

var panoViewer = new eg.view360.PanoViewer(
    container,
    {
        image: "../../img/equi.jpg",
        yaw: 45, // not mandatory, default value is 0.
        pitch: 13, // not mandatory, default value is 0.
        fov: 110 // not mandatory, default value is 65.
    }
);
